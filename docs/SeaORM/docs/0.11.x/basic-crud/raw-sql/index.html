原生SQL

通过原生SQL查询​

您可以使用适当的参数绑定语法从原生查询中选择Model，即对于MySQL和SQLite使用?，对于PostgreSQL使用$N。

let 芝士: Option<蛋糕::Model> = 蛋糕::Entity::find()
    .from_raw_sql(Statement::from_sql_and_values(
        DbBackend::Postgres,
        r#"SELECT "蛋糕"."id", "蛋糕"."name" FROM "蛋糕" WHERE "id" = $1"#,
        [1.into()],
    ))
    .one(&db)
    .await?;

您也可以选择一个自定义模型。在这里，我们从蛋糕中选择所有唯一的名字。

#[derive(Debug, FromQueryResult)]
pub struct 独特的蛋糕 {
    名字: String,
}

let 独特的: Vec<独特的蛋糕> = 独特的蛋糕::find_by_statement(Statement::from_sql_and_values(
        DbBackend::Postgres,
        r#"SELECT "蛋糕"."名字" FROM "蛋糕" GROUP BY "蛋糕"."名字""#,
        [],
    ))
    .all(&db)
    .await?;

如果您事先不知道模型的样子，可以使用JsonValue。

let 独特的: Vec<JsonValue> = JsonValue::find_by_statement(Statement::from_sql_and_values(
        DbBackend::Postgres,
        r#"SELECT "蛋糕"."名字" FROM "蛋糕" GROUP BY "蛋糕"."名字""#,
        [],
    ))
    .all(&db)
    .await?;

您可以分页SelectorRaw并批量获取Model。

let mut 蛋糕页面 = 蛋糕::Entity::find()
    .from_raw_sql(Statement::from_sql_and_values(
        DbBackend::Postgres,
        r#"SELECT "蛋糕"."id", "蛋糕"."名字" FROM "蛋糕" WHERE "id" = $1"#,
        [1.into()],
    ))
    .paginate(db, 50);
 
while let Some(蛋糕们) = 蛋糕页面.fetch_and_next().await? {
    // 对蛋糕们: Vec<蛋糕::Model>执行一些操作
}

获取原生SQL查询​

在任何CRUD操作上使用build和to_string方法来获取用于调试目的的数据库特定的原生SQL。

use sea_orm::DatabaseBackend;

assert_eq!(
    蛋糕馅料::Entity::find_by_id((6, 8))
        .build(DatabaseBackend::MySql)
        .to_string(),
    [
        "SELECT `蛋糕馅料`.`蛋糕_id`, `蛋糕馅料`.`馅料_id` FROM `蛋糕馅料`",
        "WHERE `蛋糕馅料`.`蛋糕_id` = 6 AND `蛋糕馅料`.`馅料_id` = 8",
    ].join(" ")
);

使用Raw Query & Execute接口​

您可以使用sea-query构建SQL语句，并直接在SeaORM内的DatabaseConnection接口上进行查询/执行。

使用query_one和query_all方法获取自定义结果​

let 查询结果: Option<QueryResult> = db
    .query_one(Statement::from_string(
        DatabaseBackend::MySql,
        "SELECT * FROM `蛋糕`;".to_owned(),
    ))
    .await?;
let 查询结果 = 查询结果.unwrap();
let id: i32 = 查询结果.try_get("", "id")?;

let 查询结果列表: Vec<QueryResult> = db
    .query_all(Statement::from_string(
        DatabaseBackend::MySql,
        "SELECT * FROM `蛋糕`;".to_owned(),
    ))
    .await?;

使用execute方法执行查询​

let 执行结果: ExecResult = db
    .execute(Statement::from_string(
        DatabaseBackend::MySql,
        "DROP DATABASE IF EXISTS `sea`;".to_owned(),
    ))
    .await?;
assert_eq!(执行结果.rows_affected(), 1);

执行未预编译的SQL语句​

您可以使用ConnectionTrait::execute_unprepared执行未预编译的SQL语句。

let 执行结果: ExecResult =
    db.execute_unprepared("CREATE EXTENSION IF NOT EXISTS citext").await?;
